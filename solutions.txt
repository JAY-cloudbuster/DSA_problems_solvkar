class Solution {

    public int findClosestNumber(int[] nums) {
        int minSoFar = nums[0];
        int actualValue = nums[0];
        for (int num : nums) {
            if (Math.abs(num) < Math.abs(minSoFar)) {
                minSoFar = Math.abs(num);
                actualValue = num;
            }
            else if (Math.abs(num) == Math.abs(minSoFar)) {
                minSoFar = Math.max(minSoFar, num);
                actualValue = minSoFar;
            }
        }

        return actualValue;
    }
}
---remove duplicated from the sorted array---
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if (nums.empty()) return 0;
        int k = 1;  // At least 1 unique
        for (int i = 1; i < nums.size(); ++i) {
            if (nums[i] != nums[k-1]) {
                nums[k++] = nums[i];
            }
        }
        return k;  // nums[0..k-1] unique
    }
};

class Solution {
    public int removeElement(int[] nums, int val) {
        int k = 0;  // next position for non-val element
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != val) {
                nums[k] = nums[i];  // place non-val at front
                k++;
            }
        }
        return k;  // first k elements != val
    }
}
